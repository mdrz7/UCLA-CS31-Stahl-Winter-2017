2. Consider the following program:
	#include <iostream>
	using namespace std;

	int main()
	{
	    int len;

	    cout << "Enter a number: ";
	    cin >> len;

	    for (int i = 0; i < len; i++)
	    {
		    for (int j = i+1; j < len; j++)
		    {
		        cout << " ";
		    }
		    cout << "#" << endl;
	    }
	}
Answer:
	This prints out a receding diagonal line of n pound marks starting at the nth position where n is inputted by the user all the way to the first position.

#include <iostream>

int main() {
	int len;

	std::cout << "Enter a number: ";
	std::cin >> len;

	for (int i = 0; i < len; i++) {
		int j = i + 1;
		while (j < len) {
			std::cout << " ";
			++j;
		}
		std::cout << "#" << std::endl;
	}
	return(0);
}
